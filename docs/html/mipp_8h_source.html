512static inline std::string get_back_trace()
513{
514 std::string bt_str;
515#if defined(MIPP_ENABLE_BACKTRACE) && (defined(__GNUC__) || defined(__clang__) || defined(__llvm__)) && (defined(__linux__) || defined(__linux) || defined(__APPLE__)) && !defined(__ANDROID__)
516 const int bt_max_depth = 32;
517 void *bt_array[bt_max_depth];
518
519 size_t size = backtrace(bt_array, bt_max_depth);
520 char** bt_symbs = backtrace_symbols(bt_array, size);
521
522 bt_str += "\nBacktrace:";
523 for (size_t i = 0; i < size; i++)
524 bt_str += "\n" + std::string(bt_symbs[i]);
525 free(bt_symbs);
526#endif
527
528 return bt_str;
529}
